# Autobot-token

소스코드 다운로드

    git clone https://github.com/evadelzz1/autobot_token.git


python 가상환경 activate 및 필요한 라이브러리 설치

    cd ./autobot_token

    pyenv versions

    pyenv local 3.11.6

    # pyenv install -list
    # pyenv install 3.11.6
    # pyenv local 3.11.6

    python -m venv .venv

    source .venv/bin/activate

    python -V   # 3.11.6

    pip install -r requirements.txt

코드 실행

    python main.py

python 가상환경 deactivate

    deactivate
